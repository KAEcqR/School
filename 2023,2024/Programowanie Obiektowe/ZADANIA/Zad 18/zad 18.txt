#include <iostream>
#include <cmath>
using namespace std;

int main() {
    double liczby[5];
    int i = 0;

    while (i < 5) {
        cout << "Podaj liczbe " << i + 1 << ": ";
        double input;

        cin >> input;

        if (cin.fail()) {
            cin.clear();
            cin.ignore(numeric_limits<streamsize>::max(), '\n');
        } else {
            liczby[i] = input;
            i++;
        }
    }

    double suma = 0.0;
    double iloczyn = 1.0;

    for (int j = 0; j < 5; j++) {
        suma += liczby[j];
        iloczyn *= liczby[j];
    }

    double arytmetyczna = suma / 5;
    double geometryczna = pow(iloczyn, 1.0 / 5);

    cout << "\nSrednia arytmetyczna: " << arytmetyczna << endl;
    cout << "Srednia geometryczna: " << geometryczna << endl;

    return 0;
}
