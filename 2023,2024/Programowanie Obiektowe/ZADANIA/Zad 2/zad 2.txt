#include <iostream>
using namespace std;

int main() {
    double r;
    const double PI = 3.14159;
    
    cout << "Podaj promień koła: ";
    cin >> r;
    
    
    double ppk = 4 * PI * r  *r;
    double objetosc = 4/3 * PI * r * r;
    
    cout << "Pole powierzchni kuli wynosi: " << ppk << endl;
    cout << "Obwód koła wynosi: " << objetosc << endl;

    return 0;
}
