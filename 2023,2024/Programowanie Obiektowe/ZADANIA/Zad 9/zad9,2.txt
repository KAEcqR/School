#include <iostream>
using namespace std;

int main() {
    int a, b, c;
   
    cout << "Podaj a: ";
    cin >> a;

    cout << "Podaj b: ";
    cin >> b;

    cout << "Podaj c: ";
    cin >> c;

    if (a == 0 || (b == 0) || (c == 0)) {
        cout << "Nie dzielimy przez 0" << endl;
    } else {
        double w = a / b + c / (a + b) - (a + c) / (a + b + a);
        cout << "Wynik: " << w << endl;
    }

    return 0;
}
