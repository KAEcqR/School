#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int x;
    int y;
    int parzyste = 0;
    int nieparzyste = 0;
    int ogolnie = 0;

    cout << "Podaj początkową wartość: ";
    cin >> x;

    cout << "Podaj końcową wartość: ";
    cin >> y;

    for (; x < y; x++) {
        ogolnie += x;
        if (x % 2 == 0) {
            parzyste += x;
        } else {
            nieparzyste += x;
        }
    }

    cout << "w zakresie suma parzystych to: " << parzyste << ", nieparzystych: " << nparzyste << ",wszystkich: " << ogolnie << endl;

    return 0;
}
