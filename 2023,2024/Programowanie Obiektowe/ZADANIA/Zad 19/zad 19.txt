#include <iostream>
using namespace std;

int obliczSrednia(int tab[], int n) {
    int suma = 0;
    for (int i = 0; i < n; i++) {
        suma += tab[i];
    }
    return suma / n;
}

void wypiszOceny(const char* przedmiot, int srednia) {
    cout << "Srednia z " << przedmiot << " to: ";
    if (srednia <= 1) {
        cout << "niedostateczny" << endl;
    } else if (srednia == 2) {
        cout << "dopuszczajacy" << endl;
    } else if (srednia == 3) {
        cout << "dostateczna" << endl;
    } else if (srednia == 4) {
        cout << "dobra" << endl;
    } else if (srednia == 5) {
        cout << "bardzo dobry" << endl;
    } else {
        cout << "celujacy" << endl;
    }
}

int main() {
    int tabp[] = {2, 5, 1};
    int sredniap = obliczSrednia(tabp, 3);
    wypiszOceny("jezyka polskiego", sredniap);

    int taba[] = {3, 3, 3};
    int sredniaa = obliczSrednia(taba, 3);
    wypiszOceny("jezyka angielskiego", sredniaa);

    int tabm[] = {5, 5, 6};
    int sredniam = obliczSrednia(tabm, 3);
    wypiszOceny("matematyki", sredniam);

    int tabi[] = {2, 1, 6};
    int sredniai = obliczSrednia(tabi, 3);
    wypiszOceny("informatyki", sredniai);

    return 0;
