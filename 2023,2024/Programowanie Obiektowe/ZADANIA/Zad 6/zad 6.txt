#include <iostream>
using namespace std;

int main() {
    int polski, angielski, matematyka;
    
    cout << "Ocena końcowa z polskiego: ";
    cin >> polski;
    cout << "Ocena końcowa z matematyki: ";
    cin >> matematyka;
    cout << "Ocena końcowa z angielskiego: ";
    cin >> angielski;
    
    if (polski <= 6 && angielski <= 6 && matematyka <= 6) {
        int srednia = (polski + angielski + matematyka) / 3;
        cout << "Średnia ocen wynosi: " << srednia << endl;
    } else {
        cout << "Oceny musza byc od 1 do 6." << endl;
    }

    return 0;
}
