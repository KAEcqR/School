#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int a, b, c, d;

    cout << "Podaj a: ";
    cin >> a;
    cout << "Podaj b: ";
    cin >> b;
    cout << "Podaj c: ";
    cin >> c;
    cout << "Podaj d: ";
    cin >> d;

    if (d == 0  || c == 0) {
        cout << "Nie dzielimy przez 0." << endl;
    } else {
        double w = ((a + b) / (c + d)) + (a / c) + (b / d);
        cout << "Wynik to: " << w << endl;
    }

    return 0;
}
